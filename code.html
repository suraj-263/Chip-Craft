Smart Gas & Fire Detection

Sensors 
int sensorPin = A0; 
int buzzer = 13;

void setup() {
  pinMode(sensorPin, INPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int data = analogRead(sensorPin);
  if (data > 500) {
    digitalWrite(buzzer, HIGH);
    Serial.println("Danger! Gas Detected");
  }
}
